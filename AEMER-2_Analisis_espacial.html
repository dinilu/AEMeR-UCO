Si queremos convertir coordenadas entre CRS tenemos que proyectar los datos

newcrs <- "+proj=robin +datum=WGS84"

arg_rob <- terra::project(arg, newcrs)

arg_4326 <- terra::project(arg_rob, "EPSG:4326")
par(mfrow=c(1, 3))
plot(arg)
plot(arg_rob)
plot(arg_4326)
